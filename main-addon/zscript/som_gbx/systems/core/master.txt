// The High Cheif, Master of Ceremonies, Big Boss Man, FINAL BOSS
class som_gbx_sys_core_master : som_gbx_object_base
{
	// Global Data
	som_gbx_global_data g;
	
	// HUD
	const HUD_ID = "eh-debug";
	
	void init()
	{
		g.hud.new_widget(HUD_ID);
	}
	
	// Functions
	void debug_text(String msg = "")
	{
		Let wid = g.hud.get_widget(HUD_ID);
		
		wid.draw = true;
		wid.scale = 2;
		wid.x = 8;
		wid.y = 8;
		wid.align_x = 0;
		wid.align_y = 0;
		
		wid.lines.Clear();
		wid.lines.Push(msg);
	}
	
	void map_start()
	{
		bool visited = g.map_man.map_has_visited(Level.MapName);
		
		if (!visited)
		{
			g.map_man.map_visit(Level.MapName);
		}
		
		// Call Map Start Funcs here
		g.bxp.map_start(visited);
	}	
}
